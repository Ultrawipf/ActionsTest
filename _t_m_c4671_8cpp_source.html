140 restoreEncHallMisc(miscval);

184bool TMC4671::initialize(){
185// active = true;
186// if(state == TMC_ControlState::uninitialized){
187// state = TMC_ControlState::Init_wait;
188// }
189 // Check if a TMC4671 is active and replies correctly
190 if(!pingDriver()){
191 ErrorHandler::addError(communicationError);
192 return false;
193 }
194
195 writeReg(1, 1);
196 if(readReg(0) == 0x00010000 && allowSlowSPI){
197 /* Slow down SPI if old TMC engineering sample is detected
198 * The first version has a high chance of glitches of the MSB
199 * when high spi speeds are used.
200 * This can cause problems for some operations.
201 */
202 pulseClipLed();
203
204 this->spiConfig.peripheral.BaudRatePrescaler = SPI_BAUDRATEPRESCALER_64;
205 spiPort.configurePort(&this->spiConfig.peripheral);
206 oldTMCdetected = true;
207 }
208
209 if(!oldTMCdetected){
210 this->setPidPrecision(pidPrecision);
211 }
212
213 // Write main constants
214
215 writeReg(0x64, 0); // No flux/torque
216 setPwm(0,conf.pwmcnt,conf.bbmL,conf.bbmH); // Set FOC @ 25khz but turn off pwm for now
217 setMotorType(conf.motconf.motor_type,conf.motconf.pole_pairs);
218 setPhiEtype(conf.motconf.phiEsource);
219 setup_HALL(hallconf); // Enables hall filter and masking
220
221 initAdc(conf.mdecA,conf.mdecB,conf.mclkA,conf.mclkB);
222 setAdcOffset(conf.adc_I0_offset, conf.adc_I1_offset);
223 setAdcScale(conf.adc_I0_scale, conf.adc_I1_scale);
224
225 // Initial adc calibration
226 if(!calibrateAdcOffset(150)){
227 changeState(TMC_ControlState::HardError); // ADC or shunt amp is broken!
228 enablePin.reset();
229 return false;
230 }
231 // brake res failsafe.
232// /*
233// * Single ended input raw value
234// * 0V = 0x7fff
235// * 4.7k / (360k+4.7k) Divider on old board.
236// * 1.5k / (71.5k+1.5k) 16.121 counts 60V new. 100V VM => 2V
237// * 13106 counts/V input.
238// */
239 setBrakeLimits(this->conf.hwconf.brakeLimLow,this->conf.hwconf.brakeLimHigh); // update limit from previously loaded constants or defaults
240
241 // Status mask
242 if(oldTMCdetected){
243 setStatusMask(0); // ES Version status output is broken
244 }else{
245 /*
246 * Enable adc clipping and pll errors
247 */
248 statusMask.asInt = 0;
249 statusMask.flags.adc_i_clipped = 1;
250 statusMask.flags.not_PLL_locked = 1;
251 setStatusMask(statusMask);
252 }
253
254 setPids(curPids); // Write basic pids
255
256 if(hasPower()){
257 enablePin.set();
258 setPwm(7);
259 calibrateAdcOffset(400); // Calibrate ADC again with power
260 active = true;
261 }
262 setEncoderType(conf.motconf.enctype);
263
264 // Update flags
265 readFlags(false); // Read all flags
266 // Home?
267 // Run in direction of N pulse. Enable flag/interrupt
268 //runOpenLoop(3000, 0, 5, 100);
269 initialized = true;
270 initTime = HAL_GetTick();
271 return initialized;
272}

278float TMC4671::getTemp(){
279 if(!this->conf.hwconf.temperatureEnabled){
280 return 0;
281 }
282 TMC4671HardwareTypeConf* hwconf = &conf.hwconf;
283
284 writeReg(0x03, 2);
285 int32_t adcval = ((readReg(0x02)) & 0xffff) - 0x7fff; // Center offset
286 adcval -= hwconf->adcOffset;
287 if(adcval <= 0){
288 return 0.0;
289 }
290 float r = hwconf->thermistor_R2 * (((float)43252 / (float)adcval)); //43252 equivalent ADC count if it was 3.3V and not 2.5V
291
292 // Beta
293 r = (1.0 / 298.15) + log(r / hwconf->thermistor_R) / hwconf->thermistor_Beta;
294 r = 1.0 / r;
295 r -= 273.15;
296 return r;
297

304void TMC4671::Run(){
305 // Main state machine
306 while(1){
307
308 switch(this->state){
309
310 case TMC_ControlState::Running:
311 {
312 // Check status, Temps, Everything alright?
313 uint32_t tick = HAL_GetTick();
314 if(tick - lastStatTime > 2000){ // Every 2s
315 lastStatTime = tick;
316 statusCheck();
317 // Get enable input. If tmc does not reply the result will read 0 or 0xffffffff (not possible normally)
318 uint32_t pins = readReg(0x76);
319 bool tmc_en = ((pins >> 15) & 0x01) && pins != 0xffffffff;
320 if(!tmc_en && active){ // Hardware emergency.
321 this->estopTriggered = true;
322 this->emergencyStop();
323 changeState(TMC_ControlState::HardError);
324 }
325
326 // Temperature sense
327 if(conf.hwconf.temperatureEnabled){
328 float temp = getTemp();
329 if(temp > conf.hwconf.temp_limit){
330 changeState(TMC_ControlState::OverTemp);
331 pulseErrLed();
332 }
333 }
334
335 }
336 Delay(200);
337 }
338 break;
339
340 case TMC_ControlState::Init_wait:
341 if(active && hasPower()){
342 if(HAL_GetTick() - initTime > (emergency ? 5000 : 1000)){
343 emergency = false;
344 if(!initialize()){
345 pulseErrLed();
346 }
347 }
348 }
349
350 break;
351
352 case TMC_ControlState::ABN_init:
353 ABN_init();
354 break;
355
356 case TMC_ControlState::AENC_init:
357 AENC_init();
358 break;
359
360 case TMC_ControlState::HardError:
361
362 break; // Broken
363
364 case TMC_ControlState::OverTemp:
365 this->stopMotor();
366 changeState(TMC_ControlState::HardError); // Block
367 break;
368
369 case TMC_ControlState::EncoderFinished: // Startup sequence done
370 setEncoderIndexFlagEnabled(true); // TODO
371 if(active){
372 startMotor();
373 changeState(TMC_ControlState::Running);
374 }else{
375 stopMotor();
376 laststate = TMC_ControlState::Running; // Go to running when starting again
377 }
378 break;
379
380 case TMC_ControlState::No_power:
381 if(hasPower() && !emergency){
382 changeState(laststateNopower);
383 setMotionMode(lastMotionMode,true);
384 ErrorHandler::clearError(ErrorCode::undervoltage);
385 enablePin.set();
386 }
387 pulseErrLed();
388 Delay(100); // wait a bit more
389 break;
390
391 default:
392
393 break;
394 }
395
396 // Optional update methods for safety
397
398 if(!hasPower() && state != TMC_ControlState::No_power){ // low voltage or overvoltage
399 lastMotionMode = curMotionMode;
400 laststateNopower = state;
401 ErrorHandler::addError(lowVoltageError);
402 setMotionMode(MotionMode::stop,true); // Disable tmc
403 changeState(TMC_ControlState::No_power);
404 enablePin.reset();
405 }
406
407 if(flagCheckInProgress){ // cause some delay until reenabling the status interrupt checking
408 setStatusFlags(0);
409 flagCheckInProgress = false;
410 }
411 Delay(10);
412 } // End while
413}

494void TMC4671::bangInitEnc(int16_t power){
495 if(!hasPower() || (this->conf.motconf.motor_type != MotorType::STEPPER && this->conf.motconf.motor_type != MotorType::BLDC)){ // If not stepper or bldc return
496 return;
497 }
498 blinkClipLed(50, 0);
499 PhiE lastphie = getPhiEtype();
500 MotionMode lastmode = getMotionMode();
501 setPhiE_ext(0);
502 setPhiEtype(PhiE::ext);
503 //setFluxTorque(power, 0);
504 //int32_t pos = getPos();
505
506
507
508 uint8_t phiEreg = 0;
509 uint8_t phiEoffsetReg = 0;
510 if(conf.motconf.enctype == EncoderType_TMC::abn){
511 phiEreg = 0x2A;
512 phiEoffsetReg = 0x29;
513 writeReg(0x27,0); //Zero encoder
514 }else if(conf.motconf.enctype == EncoderType_TMC::sincos || conf.motconf.enctype == EncoderType_TMC::uvw){
515 phiEreg = 0x46;
516 writeReg(0x41,0); //Zero encoder
517 writeReg(0x47,0); //Zero encoder
518 phiEoffsetReg = 0x45;
519 }
520
521 setPos(0);
522 updateReg(phiEoffsetReg, 0, 0xffff, 16); // Set phiE offset to zero
523 //setMotionMode(MotionMode::uqudext);
524
525 //Delay(100);
526 int16_t phiEpos = 0; // This is where the check starts too
527 setPhiE_ext(phiEpos);
528 // Ramp up flux
529 for(int16_t flux = 0; flux <= power; flux+=10){
530 setFluxTorque(flux, 0);
531 Delay(3);
532 }
533
534 int16_t phiE_enc = readReg(phiEreg)>>16;
535 Delay(100);
536 int16_t phiE_abn_old = 0;
537 int16_t c = 0;
538 uint16_t still = 0;
539 while(still < 30 && c++ < 4000){
540 // Wait for motor to stop moving
541 if(abs(phiE_enc - phiE_abn_old) < 200){
542 still++;
543 }else{
544 still = 0;
545 }
546 phiE_abn_old = phiE_enc;
547 phiE_enc=readReg(phiEreg)>>16;
548 Delay(10);
549 }
550
551 //Write offset
552 //int16_t phiE_abn = readReg(0x2A)>>16;
553 abnconf.phiEoffset = phiEpos-phiE_enc;
554 updateReg(phiEoffsetReg, abnconf.phiEoffset, 0xffff, 16);
555
556 setFluxTorque(0, 0);
557 setPhiE_ext(0);
558 setPhiEtype(lastphie);
559 setMotionMode(lastmode,true);
560 //setPos(pos+getPos());
561 setPos(0);
562
563 blinkClipLed(0, 0);
564}

566// Rotates motor to find min and max values of the encoder
567void TMC4671::calibrateAenc(){
568
569 // Rotate and measure min/max
570 blinkClipLed(250, 0);
571 PhiE lastphie = getPhiEtype();
572 MotionMode lastmode = getMotionMode();
573 //int32_t pos = getPos();
574 PosSelection possel = this->conf.motconf.pos_sel;
575 setPosSel(PosSelection::PhiE_openloop);
576 setPos(0);
577 // Ramp up flux
578 setFluxTorque(0, 0);
579 writeReg(0x23,0); // set phie openloop 0
580 setPhiEtype(PhiE::openloop);
581 setMotionMode(MotionMode::torque,true);
582
583 if(this->conf.motconf.motor_type == MotorType::STEPPER || this->conf.motconf.motor_type == MotorType::BLDC)
584 for(int16_t flux = 0; flux <= bangInitPower; flux+=10){
585 setFluxTorque(flux, 0);
586 Delay(10);
587 }
588
589 uint32_t minVal_0 = 0xffff, minVal_1 = 0xffff, minVal_2 = 0xffff;
590 uint32_t maxVal_0 = 0, maxVal_1 = 0, maxVal_2 = 0;
591 int32_t minpos = -0x8fff/std::max<int32_t>(1,std::min<int32_t>(this->aencconf.cpr/4,20)), maxpos = 0x8fff/std::max<int32_t>(1,std::min<int32_t>(this->aencconf.cpr/4,20));
592 uint32_t speed = std::max<uint32_t>(1,20/std::max<uint32_t>(1,this->aencconf.cpr/10));
593
594 if(this->conf.motconf.motor_type != MotorType::STEPPER && this->conf.motconf.motor_type != MotorType::BLDC){
595 speed*=10; // dc motors turn at a random speed. reduce the rotation time a bit by increasing openloop speed
596 }
597
598 runOpenLoop(bangInitPower, 0, speed, 100,true);
599
600 uint8_t stage = 0;
601 int32_t poles = conf.motconf.pole_pairs;
602 int32_t initialDirPos = 0;
603 while(stage != 3){
604 Delay(2);
605 if(getPos() > maxpos*poles && stage == 0){
606 runOpenLoop(bangInitPower, 0, -speed, 100,true);
607 stage = 1;
608 }else if(getPos() < minpos*poles && stage == 1){
609 // Scale might still be wrong... maxVal-minVal is too high. In theory its 0xffff range and scaler /256. Leave some room to prevent clipping
610 aencconf.AENC0_offset = ((maxVal_0 + minVal_0) / 2);
611 aencconf.AENC0_scale = 0xF6FF00 / (maxVal_0 - minVal_0);
612 if(conf.motconf.enctype == EncoderType_TMC::uvw){
613 aencconf.AENC1_offset = ((maxVal_1 + minVal_1) / 2);
614 aencconf.AENC1_scale = 0xF6FF00 / (maxVal_1 - minVal_1);
615 }
616
617 aencconf.AENC2_offset = ((maxVal_2 + minVal_2) / 2);
618 aencconf.AENC2_scale = 0xF6FF00 / (maxVal_2 - minVal_2);
619 aencconf.rdir = false;
620 setup_AENC(aencconf);
621 runOpenLoop(0, 0, 0, 1000,true);
622 Delay(250);
623 // Zero aenc
624 writeReg(0x41, 0);
625 initialDirPos = readReg(0x41);
626 runOpenLoop(bangInitPower, 0, speed, 100,true);
627 stage = 2;
628 }else if(getPos() > 0 && stage == 2){
629 stage = 3;
630 runOpenLoop(0, 0, 0, 1000,true);
631 }
632
633 writeReg(0x03,2);
634 uint32_t aencUX = readReg(0x02)>>16;
635 writeReg(0x03,3);
636 uint32_t aencWY_VN = readReg(0x02) ;
637 uint32_t aencWY = aencWY_VN >> 16;
638 uint32_t aencVN = aencWY_VN & 0xffff;
639
640 minVal_0 = std::min(minVal_0,aencUX);
641 minVal_1 = std::min(minVal_1,aencVN);
642 minVal_2 = std::min(minVal_2,aencWY);
643
644 maxVal_0 = std::max(maxVal_0,aencUX);
645 maxVal_1 = std::max(maxVal_1,aencVN);
646 maxVal_2 = std::max(maxVal_2,aencWY);
647 }
648 // Scale is not actually important. but offset must be perfect
649 aencconf.AENC0_offset = ((maxVal_0 + minVal_0) / 2);
650 aencconf.AENC0_scale = 0xF6FF00 / (maxVal_0 - minVal_0);
651 if(conf.motconf.enctype == EncoderType_TMC::uvw){
652 aencconf.AENC1_offset = ((maxVal_1 + minVal_1) / 2);
653 aencconf.AENC1_scale = 0xF6FF00 / (maxVal_1 - minVal_1);
654 }
655 aencconf.AENC2_offset = ((maxVal_2 + minVal_2) / 2);
656 aencconf.AENC2_scale = 0xF6FF00 / (maxVal_2 - minVal_2);
657 int32_t newDirPos = readReg(0x41);
658 aencconf.rdir = (initialDirPos - newDirPos) > 0;
659 setup_AENC(aencconf);
660 // Restore settings
661 setPhiEtype(lastphie);
662 setMotionMode(lastmode,true);
663 setPosSel(possel);
664 setPos(0);
665
666 blinkClipLed(0, 0);
667}

672bool TMC4671::checkEncoder(){
673 if(this->conf.motconf.motor_type != MotorType::STEPPER && this->conf.motconf.motor_type != MotorType::BLDC){ // If not stepper or bldc return
674 return true;
675 }
676 blinkClipLed(150, 0);
677 uint8_t phiEreg = 0;
678 if(conf.motconf.enctype == EncoderType_TMC::abn){
679 phiEreg = 0x2A;
680 }else if(conf.motconf.enctype == EncoderType_TMC::sincos || conf.motconf.enctype == EncoderType_TMC::uvw){
681 phiEreg = 0x46;
682 }
683
684 const uint16_t maxcount = 50;
685 const int16_t startAngle = 0;
686 const int16_t targetAngle = 0x3FFF;
687
688 bool result = true;
689 PhiE lastphie = getPhiEtype();
690 MotionMode lastmode = getMotionMode();
691 setFluxTorque(0, 0);
692 setPhiEtype(PhiE::ext);
693
694 setPhiE_ext(startAngle);
695 // Ramp up flux
696 for(int16_t flux = 0; flux <= bangInitPower; flux+=20){
697 setFluxTorque(flux, 0);
698 Delay(2);
699 }
700
701 //Forward
702 int16_t phiE_enc = 0;
703 uint16_t failcount = 0;
704 int16_t revCount = 0;
705 for(int16_t angle = startAngle;angle<targetAngle;angle+=0x00ff){
706 uint16_t c = 0;
707 setPhiE_ext(angle);
708 Delay(10);
709 phiE_enc = (int16_t)(readReg(phiEreg)>>16);
710 int16_t err = abs(phiE_enc - angle);
711 int16_t nErr = abs(phiE_enc + angle);
712 // Wait more
713 while(err > 2500 && nErr > 2500 && c++ < 50){
714 phiE_enc = (int16_t)(readReg(phiEreg)>>16);
715 err = abs(phiE_enc - angle);
716 nErr = abs(angle - phiE_enc);
717 Delay(10);
718 }
719 if(err > nErr){
720 revCount++;
721 }
722 if(c >= maxcount){
723 failcount++;
724 if(failcount > 10){
725 result = false;
726 break;
727 }
728 }
729 }
730
731 // Backward
732 if(result){ // Only if not already failed
733 for(int16_t angle = targetAngle;angle>startAngle;angle -= 0x00ff){
734 uint16_t c = 0;
735 setPhiE_ext(angle);
736 Delay(10);
737 phiE_enc = (int16_t)(readReg(phiEreg)>>16);
738 int16_t err = abs(phiE_enc - angle);
739 int16_t nErr = abs(phiE_enc + angle);
740 // Wait more
741 while(err > 2500 && nErr > 2500 && c++ < 50){
742 phiE_enc = (int16_t)(readReg(phiEreg)>>16);
743 err = abs(phiE_enc - angle);
744 nErr = abs(angle - phiE_enc);
745 Delay(10);
746 }
747 if(err > nErr){
748 revCount++;
749 }
750 if(c >= maxcount){
751 failcount++;
752 if(failcount > 10){
753 result = false;
754 break;
755 }
756 }
757 }
758 }
759 if(revCount > maxcount){ // Encoder seems reversed
760 // reverse encoder
761 if(this->conf.motconf.enctype == EncoderType_TMC::abn){
762 this->abnconf.rdir = !this->abnconf.rdir;
763 setup_ABN_Enc(abnconf);
764 }else if(this->conf.motconf.enctype == EncoderType_TMC::sincos || this->conf.motconf.enctype == EncoderType_TMC::uvw){
765 this->aencconf.rdir = !this->aencconf.rdir;
766 setup_AENC(aencconf);
767 }
768 }
769
770 setFluxTorque(0, 0);
771 setPhiE_ext(0);
772 setPhiEtype(lastphie);
773 setMotionMode(lastmode,true);
774
775 if(result){
776 encstate = ENC_InitState::OK;
777 }
778 blinkClipLed(0, 0);
779 return result;

849bool TMC4671::calibrateAdcOffset(uint16_t time){
850
851 uint16_t measuretime_idle = time;
852 uint32_t measurements_idle = 0;
853 uint64_t totalA=0;
854 uint64_t totalB=0;
855
856 writeReg(0x03, 0); // Read raw adc
857 PhiE lastphie = getPhiEtype();
858 MotionMode lastmode = getMotionMode();
859 setMotionMode(MotionMode::stop,true);
860
861 uint16_t lastrawA=conf.adc_I0_offset, lastrawB=conf.adc_I1_offset;
862
863 //pulseClipLed(); // Turn on led
864 // Disable drivers and measure many samples of zero current
865 //enablePin.reset();
866 uint32_t tick = HAL_GetTick();
867 while(HAL_GetTick() - tick < measuretime_idle){ // Measure idle
868 uint32_t adcraw = readReg(0x02);
869 uint16_t rawA = adcraw & 0xffff;
870 uint16_t rawB = (adcraw >> 16) & 0xffff;
871 // Signflip filter for SPI bug
872 if(abs(lastrawA-rawA) < 10000 && abs(lastrawB-rawB) < 10000){
873 totalA += rawA;
874 totalB += rawB;
875 measurements_idle++;
876 lastrawA = rawA;
877 lastrawB = rawB;
878 }
879 }
880 //enablePin.set();
881 uint32_t offsetAidle = totalA / (measurements_idle);
882 uint32_t offsetBidle = totalB / (measurements_idle);
883
884 // Check if offsets are in a valid range
885 if(totalA < 100 || totalB < 100 || (abs((int32_t)offsetAidle - 0x7fff) > 5000 || abs((int32_t)offsetBidle - 0x7fff) > 5000)){
886 ErrorHandler::addError(Error(ErrorCode::adcCalibrationError,ErrorType::critical,"TMC Adc/Shunt offset calibration failed."));
887 blinkErrLed(100, 0); // Blink forever
888 setPwm(0); //Disable pwm
889 this->changeState(TMC_ControlState::HardError);
890 return false; // An adc or shunt amp is likely broken. do not proceed.
891 }
892 conf.adc_I0_offset = offsetAidle;
893 conf.adc_I1_offset = offsetBidle;
894 setAdcOffset(conf.adc_I0_offset, conf.adc_I1_offset);
895 // ADC Offsets should now be close to perfect
896
897 setPhiEtype(lastphie);
898 setMotionMode(lastmode,true);
899 return true;

1543void TMC4671::estimateABNparams(){
1544 blinkClipLed(100, 0);
1545 int32_t pos = getPos();
1546 setPos(0);
1547 PhiE lastphie = getPhiEtype();
1548 MotionMode lastmode = getMotionMode();
1549 updateReg(0x25, 0,0x1000,12); // Set dir normal
1550 setPhiE_ext(0);
1551 setPhiEtype(PhiE::ext);
1552 setFluxTorque(0, 0);
1553 setMotionMode(MotionMode::torque,true);
1554 for(int16_t flux = 0; flux <= bangInitPower; flux+=10){
1555 setFluxTorque(flux, 0);
1556 Delay(5);
1557 }
1558
1559
1560 int16_t phiE_abn = readReg(0x2A)>>16;
1561 int16_t phiE_abn_old = 0;
1562 int16_t rcount=0,c = 0; // Count how often direction was in reverse
1563 uint16_t highcount = 0; // Count high state of n pulse for polarity estimation
1564
1565 // Rotate a bit
1566 for(int16_t p = 0;p<0x0fff;p+=0x2f){
1567 setPhiE_ext(p);
1568 Delay(10);
1569 c++;
1570 phiE_abn_old = phiE_abn;
1571 phiE_abn = readReg(0x2A)>>16;
1572 // Count how often the new position was lower than the previous indicating a reversed encoder or motor direction
1573 if(phiE_abn < phiE_abn_old){
1574 rcount++;
1575 }
1576 if((readReg(0x76) & 0x04) >> 2){
1577 highcount++;
1578 }
1579 }
1580 setPos(pos+getPos());
1581
1582 setFluxTorque(0, 0);
1583 setPhiEtype(lastphie);
1584 setMotionMode(lastmode,true);
1585
1586 bool npol = highcount > c/2;
1587 abnconf.rdir = rcount > c/2;
1588// if(npol != abnconf.npol) // Invert dir if polarity was reversed TODO correct? likely wrong at the moment
1589// abnconf.rdir = !abnconf.rdir;
1590
1591 abnconf.apol = npol;
1592 abnconf.bpol = npol;
1593 abnconf.npol = npol;
1594 blinkClipLed(0, 0);
1595}

1732 statusFlags = readFlags(); // Update current flags
1733
1734 // encoder index flag was set since last check. Check if the flag matching the current encoder is set
1735 if( (statusFlags.flags.ENC_N && this->conf.motconf.enctype == EncoderType_TMC::abn) || (statusFlags.flags.AENC_N && this->conf.motconf.enctype == EncoderType_TMC::sincos) ){
1736 encoderIndexHit();
1737 }
1738

1740 // Critical error. PLL not locked
1741 ErrorHandler::addError(Error(ErrorCode::tmcPLLunlocked, ErrorType::critical, "TMC PLL not locked"));
1742 }
1743
1744
1745 setStatusFlags(0); // Reset flags
1746 if(readFlags().asInt != statusFlags.asInt){ // Condition is cleared. if not we will reset it in the main loop later to get out of the isr and cause some delay
1747 flagCheckInProgress = false;
1748 }
