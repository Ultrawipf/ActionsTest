170//--------------------------------------------------------------------+
171// CLASS-USBD API (don't require to verify parameters)
172//--------------------------------------------------------------------+
173bool tuh_init(void)
174{
175 tu_memclr(_usbh_devices, sizeof(usbh_device_t)*(CFG_TUSB_HOST_DEVICE_MAX+1));
176
177 //------------- Enumeration & Reporter Task init -------------//
178 _usbh_q = osal_queue_create( &_usbh_qdef );
179 TU_ASSERT(_usbh_q != NULL);
180
181 //------------- Semaphore, Mutex for Control Pipe -------------//
182 for(uint8_t i=0; i<CFG_TUSB_HOST_DEVICE_MAX+1; i++) // including address zero
183 {
184 usbh_device_t * const dev = &_usbh_devices[i];
185
186#if CFG_TUSB_OS != OPT_OS_NONE
187 dev->mutex = osal_mutex_create(&dev->mutexdef);
188 TU_ASSERT(dev->mutex);
189#endif
190
191 memset(dev->itf2drv, 0xff, sizeof(dev->itf2drv)); // invalid mapping
192 memset(dev->ep2drv , 0xff, sizeof(dev->ep2drv )); // invalid mapping
193 }
194
195 // Class drivers init
196 for (uint8_t drv_id = 0; drv_id < USBH_CLASS_DRIVER_COUNT; drv_id++)
197 {
198 TU_LOG2("%s init\r\n", usbh_class_drivers[drv_id].name);
199 usbh_class_drivers[drv_id].init();
200 }
201
202 TU_ASSERT(hcd_init(TUH_OPT_RHPORT));
203 hcd_int_enable(TUH_OPT_RHPORT);
204
205 return true;
206}

208bool usbh_edpt_claim(uint8_t dev_addr, uint8_t ep_addr)
209{
210 uint8_t const epnum = tu_edpt_number(ep_addr);
211 uint8_t const dir = tu_edpt_dir(ep_addr);
212
213 usbh_device_t* dev = &_usbh_devices[dev_addr];
214
215#if CFG_TUSB_OS != OPT_OS_NONE
216 // pre-check to help reducing mutex lock
217 TU_VERIFY((dev->ep_status[epnum][dir].busy == 0) && (dev->ep_status[epnum][dir].claimed == 0));
218 osal_mutex_lock(dev->mutex, OSAL_TIMEOUT_WAIT_FOREVER);
219#endif
220
221 // can only claim the endpoint if it is not busy and not claimed yet.
222 bool const ret = (dev->ep_status[epnum][dir].busy == 0) && (dev->ep_status[epnum][dir].claimed == 0);
223 if (ret)
224 {
225 dev->ep_status[epnum][dir].claimed = 1;
226 }
227
228#if CFG_TUSB_OS != OPT_OS_NONE
229 osal_mutex_unlock(dev->mutex);
230#endif
231
232 return ret;
233}
234
235bool usbh_edpt_release(uint8_t dev_addr, uint8_t ep_addr)
236{
237 uint8_t const epnum = tu_edpt_number(ep_addr);
238 uint8_t const dir = tu_edpt_dir(ep_addr);
239
240 usbh_device_t* dev = &_usbh_devices[dev_addr];
241
242#if CFG_TUSB_OS != OPT_OS_NONE
243 osal_mutex_lock(dev->mutex, OSAL_TIMEOUT_WAIT_FOREVER);
244#endif
245
246 // can only release the endpoint if it is claimed and not busy
247 bool const ret = (dev->ep_status[epnum][dir].busy == 0) && (dev->ep_status[epnum][dir].claimed == 1);
248 if (ret)
249 {
250 dev->ep_status[epnum][dir].claimed = 0;
251 }
252
253#if CFG_TUSB_OS != OPT_OS_NONE
254 osal_mutex_unlock(dev->mutex);
255#endif
256
257 return ret;
258}

281bool usbh_edpt_open(uint8_t rhport, uint8_t dev_addr, tusb_desc_endpoint_t const * ep_desc)
282{
283 bool ret = hcd_edpt_open(rhport, dev_addr, ep_desc);
284
285 if (ret)
286 {
287 usbh_device_t* dev = &_usbh_devices[dev_addr];
288
289 // new endpoints belongs to latest interface (last valid value)
290 // TODO FIXME not true with ISO
291 uint8_t drvid = 0xff;
292 for(uint8_t i=0; i < sizeof(dev->itf2drv); i++)
293 {
294 if ( dev->itf2drv[i] == 0xff ) break;
295 drvid = dev->itf2drv[i];
296 }
297 TU_ASSERT(drvid < USBH_CLASS_DRIVER_COUNT);
298
299 uint8_t const ep_addr = ep_desc->bEndpointAddress;
300 dev->ep2drv[tu_edpt_number(ep_addr)][tu_edpt_dir(ep_addr)] = drvid;
301 }
302
303 return ret;
304}

368// a device unplugged on hostid, hub_addr, hub_port
369// return true if found and unmounted device, false if cannot find
370static void usbh_device_unplugged(uint8_t rhport, uint8_t hub_addr, uint8_t hub_port)
371{
372 //------------- find the all devices (star-network) under port that is unplugged -------------//
373 for (uint8_t dev_addr = 0; dev_addr <= CFG_TUSB_HOST_DEVICE_MAX; dev_addr ++)
374 {
375 usbh_device_t* dev = &_usbh_devices[dev_addr];
376
377 // TODO Hub multiple level
378 if (dev->rhport == rhport &&
379 (hub_addr == 0 || dev->hub_addr == hub_addr) && // hub_addr == 0 & hub_port == 0 means roothub
380 (hub_port == 0 || dev->hub_port == hub_port) &&
381 dev->state != TUSB_DEVICE_STATE_UNPLUG)
382 {
383 // Invoke callback before close driver
384 if (tuh_umount_cb) tuh_umount_cb(dev_addr);
385
386 // Close class driver
387 for (uint8_t drv_id = 0; drv_id < USBH_CLASS_DRIVER_COUNT; drv_id++)
388 {
389 TU_LOG2("%s close\r\n", usbh_class_drivers[drv_id].name);
390 usbh_class_drivers[drv_id].close(dev_addr);
391 }
392
393 memset(dev->itf2drv, 0xff, sizeof(dev->itf2drv)); // invalid mapping
394 memset(dev->ep2drv , 0xff, sizeof(dev->ep2drv )); // invalid mapping
395
396 hcd_device_close(rhport, dev_addr);
397
398 dev->state = TUSB_DEVICE_STATE_UNPLUG;
399 }
400 }
401}

403/* USB Host Driver task
404 * This top level thread manages all host controller event and delegates events to class-specific drivers.
405 * This should be called periodically within the mainloop or rtos thread.
406 *
407 @code
408 int main(void)
409 {
410 application_init();
411 tusb_init();
412
413 while(1) // the mainloop
414 {
415 application_code();
416 tuh_task(); // tinyusb host task
417 }
418 }
419 @endcode
420 */
421void tuh_task(void)
422{
423 // Skip if stack is not initialized
424 if ( !tusb_inited() ) return;
425
426 // Loop until there is no more events in the queue
427 while (1)
428 {
429 hcd_event_t event;
430 if ( !osal_queue_receive(_usbh_q, &event) ) return;
431
432 switch (event.event_id)
433 {
434 case HCD_EVENT_DEVICE_ATTACH:
435 // TODO due to the shared _usbh_ctrl_buf, we must complete enumerating
436 // one device before enumerating another one.
437 TU_LOG2("USBH DEVICE ATTACH\r\n");
438 enum_new_device(&event);
439 break;
440
441 case HCD_EVENT_DEVICE_REMOVE:
442 TU_LOG2("USBH DEVICE REMOVED\r\n");
443 usbh_device_unplugged(event.rhport, event.connection.hub_addr, event.connection.hub_port);
444
445 #if CFG_TUH_HUB
446 // TODO remove
447 if ( event.connection.hub_addr != 0)
448 {
449 // done with hub, waiting for next data on status pipe
450 (void) hub_status_pipe_queue( event.connection.hub_addr );
451 }
452 #endif
453 break;
454
455 case HCD_EVENT_XFER_COMPLETE:
456 {
457 usbh_device_t* dev = &_usbh_devices[event.dev_addr];
458 uint8_t const ep_addr = event.xfer_complete.ep_addr;
459 uint8_t const epnum = tu_edpt_number(ep_addr);
460 uint8_t const ep_dir = tu_edpt_dir(ep_addr);
461
462 TU_LOG2("on EP %02X with %u bytes\r\n", ep_addr, (unsigned int) event.xfer_complete.len);
463
464 if ( 0 == epnum )
465 {
466 usbh_control_xfer_cb(event.dev_addr, ep_addr, event.xfer_complete.result, event.xfer_complete.len);
467 }else
468 {
469 uint8_t drv_id = dev->ep2drv[epnum][ep_dir];
470 TU_ASSERT(drv_id < USBH_CLASS_DRIVER_COUNT, );
471
472 TU_LOG2("%s xfer callback\r\n", usbh_class_drivers[drv_id].name);
473 usbh_class_drivers[drv_id].xfer_cb(event.dev_addr, ep_addr, event.xfer_complete.result, event.xfer_complete.len);
474 }
475 }
476 break;
477
478 case USBH_EVENT_FUNC_CALL:
479 if ( event.func_call.func ) event.func_call.func(event.func_call.param);
480 break;
481
482 default: break;
483 }
484 }
485}

499void usbh_driver_set_config_complete(uint8_t dev_addr, uint8_t itf_num)
500{
501 usbh_device_t* dev = &_usbh_devices[dev_addr];
502
503 for(itf_num++; itf_num < sizeof(dev->itf2drv); itf_num++)
504 {
505 // continue with next valid interface
506 uint8_t const drv_id = dev->itf2drv[itf_num];
507 if (drv_id != 0xff)
508 {
509 usbh_class_driver_t const * driver = &usbh_class_drivers[drv_id];
510 TU_LOG2("%s set config itf = %u\r\n", driver->name, itf_num);
511 driver->set_config(dev_addr, itf_num);
512 break;
513 }
514 }
515
516 // all interface are configured
517 if (itf_num == sizeof(dev->itf2drv))
518 {
519 // Invoke callback if available
520 if (tuh_mount_cb) tuh_mount_cb(dev_addr);
521 }
522}

566static bool enum_hub_get_status1_complete(uint8_t dev_addr, tusb_control_request_t const * request, xfer_result_t result)
567{
568 (void) dev_addr; (void) request;
569 TU_ASSERT(XFER_RESULT_SUCCESS == result);
570 usbh_device_t* dev0 = &_usbh_devices[0];
571
572 hub_port_status_response_t port_status;
573 memcpy(&port_status, _usbh_ctrl_buf, sizeof(hub_port_status_response_t));
574
575 // Acknowledge Port Reset Change if Reset Successful
576 if (port_status.change.reset)
577 {
578 TU_ASSERT( hub_port_clear_feature(dev0->hub_addr, dev0->hub_port, HUB_FEATURE_PORT_RESET_CHANGE, enum_hub_clear_reset1_complete) );
579 }
580
581 return true;
582}

584static bool enum_hub_get_status0_complete(uint8_t dev_addr, tusb_control_request_t const * request, xfer_result_t result)
585{
586 (void) dev_addr; (void) request;
587 TU_ASSERT(XFER_RESULT_SUCCESS == result);
588 usbh_device_t* dev0 = &_usbh_devices[0];
589
590 hub_port_status_response_t port_status;
591 memcpy(&port_status, _usbh_ctrl_buf, sizeof(hub_port_status_response_t));
592
593 if ( !port_status.status.connection )
594 {
595 // device unplugged while delaying, nothing else to do, queue hub status
596 return hub_status_pipe_queue(dev_addr);
597 }
598
599 dev0->speed = (port_status.status.high_speed) ? TUSB_SPEED_HIGH :
600 (port_status.status.low_speed ) ? TUSB_SPEED_LOW : TUSB_SPEED_FULL;
601
602 // Acknowledge Port Reset Change
603 if (port_status.change.reset)
604 {
605 hub_port_clear_feature(dev0->hub_addr, dev0->hub_port, HUB_FEATURE_PORT_RESET_CHANGE, enum_hub_clear_reset0_complete);
606 }
607
608 return true;
609}

612static bool enum_request_set_addr(void)
613{
614 // Set Address
615 TU_LOG2("Set Address \r\n");
616 uint8_t const new_addr = get_new_address();
617 TU_ASSERT(new_addr <= CFG_TUSB_HOST_DEVICE_MAX); // TODO notify application we reach max devices
618
619 usbh_device_t* dev0 = &_usbh_devices[0];
620 usbh_device_t* new_dev = &_usbh_devices[new_addr];
621
622 new_dev->rhport = dev0->rhport;
623 new_dev->hub_addr = dev0->hub_addr;
624 new_dev->hub_port = dev0->hub_port;
625 new_dev->speed = dev0->speed;
626 new_dev->connected = 1;
627 new_dev->ep0_packet_size = ((tusb_desc_device_t*) _usbh_ctrl_buf)->bMaxPacketSize0;
628
629 tusb_control_request_t const new_request =
630 {
631 .bmRequestType_bit =
632 {
633 .recipient = TUSB_REQ_RCPT_DEVICE,
634 .type = TUSB_REQ_TYPE_STANDARD,
635 .direction = TUSB_DIR_OUT
636 },
637 .bRequest = TUSB_REQ_SET_ADDRESS,
638 .wValue = new_addr,
639 .wIndex = 0,
640 .wLength = 0
641 };
642
643 TU_ASSERT( tuh_control_xfer(0, &new_request, NULL, enum_set_address_complete) );
644
645 return true;
646}

648static bool enum_new_device(hcd_event_t* event)
649{
650 usbh_device_t* dev0 = &_usbh_devices[0];
651 dev0->rhport = event->rhport; // TODO refractor integrate to device_pool
652 dev0->hub_addr = event->connection.hub_addr;
653 dev0->hub_port = event->connection.hub_port;

906static bool parse_configuration_descriptor(uint8_t dev_addr, tusb_desc_configuration_t const* desc_cfg)
907{
908 usbh_device_t* dev = &_usbh_devices[dev_addr];
909 uint8_t const* p_desc = (uint8_t const*) desc_cfg;
910 p_desc = tu_desc_next(p_desc);
911
912 // parse each interfaces
913 while( p_desc < _usbh_ctrl_buf + desc_cfg->wTotalLength )
914 {
915 // skip until we see interface descriptor
916 if ( TUSB_DESC_INTERFACE != tu_desc_type(p_desc) )
917 {
918 p_desc = tu_desc_next(p_desc); // skip the descriptor, increase by the descriptor's length
919 }else
920 {
921 tusb_desc_interface_t const* desc_itf = (tusb_desc_interface_t const*) p_desc;
922
923 // Check if class is supported
924 uint8_t drv_id;
925 for (drv_id = 0; drv_id < USBH_CLASS_DRIVER_COUNT; drv_id++)
926 {
927 if ( usbh_class_drivers[drv_id].class_code == desc_itf->bInterfaceClass ) break;
928 }
929
930 if( drv_id >= USBH_CLASS_DRIVER_COUNT )
931 {
932 // skip unsupported class
933 p_desc = tu_desc_next(p_desc);
934 }
935 else
936 {
937 usbh_class_driver_t const * driver = &usbh_class_drivers[drv_id];
938
939 // Interface number must not be used already TODO alternate interface
940 TU_ASSERT( dev->itf2drv[desc_itf->bInterfaceNumber] == 0xff );
941 dev->itf2drv[desc_itf->bInterfaceNumber] = drv_id;
942
943 if (desc_itf->bInterfaceClass == TUSB_CLASS_HUB && dev->hub_addr != 0)
944 {
945 // TODO Attach hub to Hub is not currently supported
946 // skip this interface
947 p_desc = tu_desc_next(p_desc);
948 }
949 else
950 {
951 TU_LOG2("%s open\r\n", driver->name);
952
953 uint16_t itf_len = 0;
954 TU_ASSERT( driver->open(dev->rhport, dev_addr, desc_itf, &itf_len) );
955 TU_ASSERT( itf_len >= sizeof(tusb_desc_interface_t) );
956 p_desc += itf_len;
957 }
958 }
959 }
960 }
961
962 return true;
963}
