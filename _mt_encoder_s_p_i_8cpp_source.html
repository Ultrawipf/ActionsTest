88void MtEncoderSPI::updateAngleStatus(){
89 uint8_t txbuf[4] = {0x03 | MAGNTEK_READ,0,0,0};
90 uint8_t rxbuf[4] = {0,0,0,0};
91 spiPort.transmitReceive(txbuf, rxbuf, 4, this, 1000);
92
93 uint32_t angle17_10 = rxbuf[1];
94 uint32_t angle9_4 = rxbuf[2];
95 uint32_t angle3_0 = rxbuf[3];
96
97 nomag = (angle9_4 & 0x02) >> 1;
98 overspeed = (angle3_0 & 0x04) >> 2;
99 angle9_4 = (angle9_4 & 0xFC) >> 2;
100 angle3_0 = (angle3_0 & 0xE0) >> 5;
101
102 curAngleInt = (angle3_0) | (angle9_4 << 4) | (angle17_10 << 10);
103
104 if(lastAngleInt < 0x10000 && curAngleInt > 0x30000){ // Underflowed
105 rotations--;
106 }
107 else if(curAngleInt < 0x10000 && lastAngleInt > 0x30000){ // Overflowed
108 rotations++;
109 }
110
111 curPos = rotations * getCpr() + curAngleInt;
112 lastAngleInt = curAngleInt;
113}
