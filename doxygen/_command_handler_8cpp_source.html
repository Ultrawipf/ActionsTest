38std::string CommandHandler::getCommandsHelpstring(){

41 if(info.name == nullptr || cmdHandlerInfo.clsname == nullptr){
42 return "";
43 }
44 helpstring.append(info.name);
45 helpstring += "(";

47 helpstring += "." + std::to_string(cmdHandlerInfo.instance) + "):\n";

51 if(!handlerHelp.empty()){
52 helpstring += handlerHelp + "\n";

54
55 if(registeredCommands.empty()){
56 helpstring += "No commands.";

58 helpstring += "Commands with help:\n";

61 helpstring += std::string(cmd.cmd) + ":\t" + std::string(cmd.helpstring);
62 if(cmd.flags & CMDFLAG_DEBUG){
63 helpstring+= " (DEBUG MODE ONLY)";
64 }
65 helpstring += "\n";
66 }
67
68 }
69 }

172CommandStatus CommandHandler::internalCommand(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
173 CommandStatus result = CommandStatus::OK;
174 switch(static_cast<CommandHandlerCommands>(cmd.cmdId)){
175
176 case CommandHandlerCommands::id:
177 replies.push_back(CommandReply(this->getInfo().id));
178 break;
179
180 case CommandHandlerCommands::name:
181 replies.push_back(CommandReply(this->getInfo().name));
182 break;
183
184 case CommandHandlerCommands::help:
185 if(cmd.type == CMDtype::info){
186 replies.push_back(CommandReply(this->getCsvHelpstring()));
187 }else{
188 replies.push_back(CommandReply(this->getCommandsHelpstring()));
189 }
190 break;
191
192 case CommandHandlerCommands::cmdhandleruid:
193 replies.push_back(CommandReply(this->getCommandHandlerID()));
194 break;
195
196 case CommandHandlerCommands::instance:
197 replies.push_back(CommandReply(this->getCommandHandlerInfo()->instance));
198 break;
199
200 case CommandHandlerCommands::selectionid:
201 replies.push_back(CommandReply(this->getSelectionID()));
202 break;
203
204 default:
205 result = CommandStatus::NOT_FOUND;
206 break;
207 }
208 return result;
209}

417void CommandHandler::addCommandHandler(){
418 cmdHandlerListMutex.Lock();
419 while(this->cmdHandlerInfo.commandHandlerID != 0xffff){
420 auto res = CommandHandler::cmdHandlerIDs.insert(this->cmdHandlerInfo.commandHandlerID);
421 if(res.second){ // Element was inserted
422 break;
423 }
424 this->cmdHandlerInfo.commandHandlerID++; // Try next id
425 }
426 addCallbackHandler(cmdHandlers, this);
427 cmdHandlerListMutex.Unlock();
428}
