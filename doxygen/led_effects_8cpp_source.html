25void blinkLed(Ledstruct* led,uint16_t period,uint16_t blinks){
26 // Stop blinking
27 if(period == 0 && blinks == 0){
28 led->blinks = 0;
29 led->period = 0;
30 HAL_GPIO_WritePin(led->port, led->pin, GPIO_PIN_RESET);
31 }else{
32 led->blinks = (blinks * 2) -1;
33 led->period = period;
34 led->tick = HAL_GetTick();
35 HAL_GPIO_WritePin(led->port, led->pin, GPIO_PIN_SET);
36 }
37}

66void updateLed(Ledstruct* led){
67 // If led has an effect (period != 0) and time is up do something
68 if(led->period != 0 && HAL_GetTick() > led->tick+led->period){
69 if(led->blinks == 0){ // No blinks left. turn off
70 HAL_GPIO_WritePin(led->port, led->pin, GPIO_PIN_RESET);
71 led->period = 0;
72 }else{
73 led->tick = HAL_GetTick();
74 HAL_GPIO_TogglePin(led->port, led->pin);
75 }
76 // If positive decrement blink counter.
77 if(led->blinks > 0){
78 led->blinks--;
79 }
80 }
81}
