109void FFBWheel::update(){
110 if(control.request_update_disabled) {
111 //logSerial("request update disabled");
112 control.update_disabled = true;
113 control.request_update_disabled = false;
114
115 }
116 if(control.update_disabled){
117 //logSerial("Update disabled");
118 return;
119 }
120 if(control.emergency){
121 pulseErrLed();
122 return;
123 }
124 if(control.resetEncoder){
125 control.resetEncoder = false;
126 axes_manager->resetPosZero();
127 }
128
129
130
131 // TODO Emulate a SOF timer...
132 if(HAL_GetTick() - lastUsbReportTick > 0 && !control.usb_disabled){
133 lastUsbReportTick = HAL_GetTick();
134 control.usb_update_flag = true;
135 }
136
137 // If either usb or timer triggered
138 if(control.usb_update_flag || control.update_flag){
139 axes_manager->update();
140 control.update_flag = false;
141 if(control.usb_update_flag){
142 control.usb_update_flag = false;
143 if(++report_rate_cnt >= usb_report_rate){
144 report_rate_cnt = 0;
145 this->send_report();
146 }
147 }
148 axes_manager->updateTorque();
149 }
150}

225void FFBWheel::send_report(){
226
227 // Read buttons
228 reportHID.buttons = 0; // Reset buttons
229 uint8_t shift = 0;
230 if(btns.size() != 0){
231 for(auto &btn : btns){
232 uint64_t buf = 0;
233 uint8_t amount = btn->readButtons(&buf);
234 reportHID.buttons |= buf << shift;
235 shift += amount;
236 }
237 }
238
239
240 // Encoder
241 //axes_manager->addAxesToReport(analogAxesReport, &count);
242
243 std::vector<int32_t>* axes = axes_manager->getAxisValues();
244 uint8_t count = 0;
245 for(auto val : *axes){
246 setHidReportAxis(&reportHID,count++,val);
247 }
248
249 // Fill remaining values with analog inputs
250 for(auto &ain : analog_inputs){
251 std::vector<int32_t>* axes = ain->getAxes();
252 for(int32_t val : *axes){
253 if(count >= analogAxisCount)
254 break;
255 setHidReportAxis(&reportHID,count++,val);
256 }
257 } // Fill rest
258 for(;count<analogAxisCount; count++){
259 setHidReportAxis(&reportHID,count,0);
260 }
261
262 /*
263 * Only send a new report if actually changed since last time or timeout and hid is ready
264 */
265 if( (reportSendCounter++ > 100/usb_report_rate || (memcmp(&lastReportHID,&reportHID,sizeof(reportHID_t)) != 0) ) && tud_hid_n_ready(0)){
266 tud_hid_report(0, reinterpret_cast<uint8_t*>(&reportHID), sizeof(reportHID_t));
267 lastReportHID = reportHID;
268 reportSendCounter = 0;
269 }
270
271}
